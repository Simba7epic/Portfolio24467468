def is_safe_ast(source: str) -> bool:
    """
    Parse the source into an AST and reject any nodes that could
    execute system calls, imports, or arbitrary expressions.
    """

        tree = ast.parse(source)

        logger.warning("Syntax error in file: %s", e)
        return False

        ast.Import,          # import os
        ast.ImportFrom,      # from os import ...
        ast.Call,            # any function call (can be relaxed selectively)
        ast.Attribute,       # obj.method access (e.g., os.system)
        ast.Global,          # global variable manipulation
        ast.Nonlocal,
    )

    for node in ast.walk(tree):
        if isinstance(node, FORBIDDEN_NODE_TYPES):
            logger.warning("Forbidden AST node detected: %s", type(node).__name__)
            return False

    return True
